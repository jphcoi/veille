            private var linkStrokeMin:Number = 1;

            private var linkLift:Number = 0;
            private var linkMin:Number;
            private var linkMax:Number;
            
          
            


            private function onJSONLoad(event:ResultEvent) : void {
                // convert the raw JSON data to a string
                var rawData:String = String(event.result);
            
                // use the JSON Library to convert the data into actionscript objects.
                var aData:Array = (JSON.decode(rawData) as Array);
            
                // set the data into an arrayCollection 
                keywordData = new ArrayCollection(aData);
                
                setup();
            }
            
            private function setup() : void {
                // add items
                for each (var obj:Object in keywordData) {
                    var item: Item = new Item(obj.keyword);
                    item.data = obj.count;
                    dataCounts.push(obj.count);
                    myGraph.add(item);
                }
                
                // sort the counts array
                dataCounts.sort(Array.NUMERIC);
                dataMin = dataCounts[0];
                dataMax = dataCounts[dataCounts.length-1];
                difference = Math.log(dataMax - dataMin);
                steps = fontMax-fontMin;
                fontLift = difference/steps;
                
                
                // setup links
                for each (var lObj:Object in keywordData) {
                    var linkItem: Item = myGraph.find(lObj.keyword);
                    var keywordLinks:Object = lObj.links;
                    var aLinks:Array = new Array();
                    for (var l:Object in keywordLinks) {
                        aLinks.push(keywordLinks[l]);
                    }
                    aLinks.sort(Array.NUMERIC);
                    
                    var linkMin:Number = aLinks[0];
                    var linkMax:Number = aLinks[aLinks.length-1];
                    if (linkMin==linkMax)
                    {
                        linkMin=0;
                    }
                    var linkdifference:Number = linkMax - linkMin;
                    var linkSteps:Number = linkStrokeMax-linkStrokeMin;
                    var linkLift:Number = linkdifference/linkSteps;
                    
                    for (var link:String in keywordLinks) {
                        var linkData:Object = new Object();
                        linkData.settings = {thickness:getLinkSize(keywordLinks[link],linkLift,linkMin), color:0xFFC051, alpha:0.9};
                        switch(linkData.settings.thickness) {
        case 10:
                                linkData.settings.color = 0xFFC051;
                                linkData.settings.thickness=linkData.settings.thickness+1;
                                break;
        case 9:
                                linkData.settings.color = 0xFFC051;
                                linkData.settings.thickness=linkData.settings.thickness+1;
                                break;
        
        case 8:
                                linkData.settings.color = 0xFFC051;
                                linkData.settings.thickness=linkData.settings.thickness+1;
                                break;
        case 7:
                                linkData.settings.color = 0xFFC051;
                                linkData.settings.thickness=linkData.settings.thickness+1;
                                break;
    case 6:
                                linkData.settings.color = 0xFFC051;
                                linkData.settings.thickness=linkData.settings.thickness+1;
                                break;
                            case 5:
                                linkData.settings.color = 0xFFC051;
                                linkData.settings.thickness=linkData.settings.thickness+1;
                                break;

                            case 4:
                            linkData.settings.color = 0xFFC051;
                            linkData.settings.thickness=linkData.settings.thickness+1;
                                    break;
                            case 3:
                            linkData.settings.color = 0xFFC051;
                                linkData.settings.thickness=linkData.settings.thickness+1;
                                    break;
                            case 2:
                                    linkData.settings.color = 0xFFC051;
                                linkData.settings.thickness=linkData.settings.thickness+1;
                                    break;
                            case 1:
                                    linkData.settings.color = 0xFFC051;
                                linkData.settings.thickness=linkData.settings.thickness+1;
                                    break;
                            case 0:
                                    
                                    linkData.settings.color = 0xFFC051;
                                linkData.settings.thickness=linkData.settings.thickness+1;
                                break;
                            case -1:
                            linkData.settings.thickness=linkData.settings.thickness+1;
                                        
                                    linkData.settings.color = 0xFFC051;
                                    linkData.settings.alpha = 0.01;
                                    
                                break;

                        }
                        if (linkData.settings.thickness > -0.1) {
                            linkData.settings.thickness=linkData.settings.thickness/1.4;
                            myGraph.link(linkItem, myGraph.find(link), linkData);
                        }
                    }
                }
                
                var factory:ClassFactory = new ClassFactory(KeywordRenderer)
                factory.properties = {fontMin:fontMin, fontMax:fontMax, fontLift:fontLift, dataMin:dataMin, dataMax:dataMax};
                mySpringGraph.itemRenderer = factory; 
                mySpringGraph.dataProvider = myGraph;
               
            }
            
            public function getLinkSize(data:Number, linkLift:Number, linkMin:Number) : Number {
                var linkStroke:Number = ((data - linkMin) / linkLift);
                if (data==0.1)
                    {return -1;}
                if (linkStroke < linkStrokeMin) {
                    return Math.round(linkStrokeMin);
                } else if (linkStroke > linkStrokeMax) {
                    return Math.round(linkStrokeMax);
                } else {
                    return Math.round(linkStroke);    
                }
            }
