<?xml version="1.0" encoding="utf-8"?>
<mx:Application 
    xmlns:mx="http://www.adobe.com/2006/mxml"
    xmlns:sg="com.adobe.flex.extras.controls.springgraph.*" 
    layout="absolute"
    creationComplete="service.send();" viewSourceURL="srcview/index.html">
    
    <mx:HTTPService 
        id="service" 
        resultFormat="text"
        url="json_data.txt"
        result="onJSONLoad(event)" />
        
    <mx:Script>
        <![CDATA[
            import mx.collections.ArrayCollection;
            import mx.rpc.events.ResultEvent;
            import com.adobe.serialization.json.*;
            import com.adobe.flex.extras.controls.springgraph.Item;
            import com.adobe.flex.extras.controls.springgraph.Graph;
            
            [Bindable]
            private var keywordData : ArrayCollection;
            private var myGraph: Graph = new Graph();
            
            private var dataCounts:Array = new Array();
            private var dataMin:Number;
            private var dataMax:Number;
            private var difference:Number;
            private var fontMin:Number = 8;
            private var fontMax:Number = 48;
            private var steps:Number = 0;
            private var fontLift:Number = 0;
            
            private function onJSONLoad(event:ResultEvent) : void {
                // convert the raw JSON data to a string
                var rawData:String = String(event.result);
            
                // use the JSON Library to convert the data into actionscript objects.
                var aData:Array = (JSON.decode(rawData) as Array);
            
                // set the data into an arrayCollection 
                keywordData = new ArrayCollection(aData);
                
                setup();
            }
            
            private function setup() : void {
                // add items
                for each (var obj:Object in keywordData) {
                    var item: Item = new Item(obj.keyword);
                    item.data = obj.count;
                    dataCounts.push(obj.count);
                    myGraph.add(item);
                }
                
                // sort the counts array
                dataCounts.sort(Array.NUMERIC);
                dataMin = dataCounts[0];
                dataMax = dataCounts[dataCounts.length-1];
                difference = dataMax - dataMin;
                steps = fontMax-fontMin;
                fontLift = difference/steps;
                
                // setup links
                for each (var lObj:Object in keywordData) {
                    var linkItem: Item = myGraph.find(lObj.keyword);
                    var links:Object = lObj.links;
                    for each (var link:String in links) {
                        myGraph.link(linkItem, myGraph.find(link));
                    }
                }
                
                mySpringGraph.dataProvider = myGraph;
            }
            
            public function getFontSize(data:Number) : Number {
                var fontSize:Number = (data - dataMin) / fontLift;
                if (fontSize < fontMin) {
                    return fontMin;
                } else if (fontSize > fontMax) {
                    return fontMax;
                } else {
                    return fontSize;    
                }
            }
            
        ]]>
    </mx:Script>    
    
    <sg:SpringGraph 
        id="mySpringGraph"
        backgroundColor="#FFFFFF" 
        lineColor="#6666ff" 
        repulsionFactor="5"
        right="10" left="10" bottom="10" top="10">
        <sg:itemRenderer>
            <mx:Component>
                <mx:VBox 
                    cornerRadius="5" 
                    backgroundColor="0x444444" 
                    paddingBottom="5" 
                    paddingLeft="5" 
                    paddingRight="5" 
                    paddingTop="5">
                    <mx:Label 
                        text="{data.id}" 
                        fontSize="{parentDocument.getFontSize(data.data)}" 
                        color="0xFFFFFF" 
                        textAlign="center" />
                </mx:VBox>
            </mx:Component>
        </sg:itemRenderer>
    </sg:SpringGraph>
    
</mx:Application>
